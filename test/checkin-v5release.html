async installPlugin(plugin) { const progressEl = this.ensurePluginProgressEl(plugin.plugin_id); const pluginItem = progressEl?.closest('.plugin-item'); const installBtn = pluginItem?.querySelector('[data-action="install"]'); try { this.setPluginProgress(plugin.plugin_id, 10, '正在准备安装...'); if (installBtn) { installBtn.disabled = true; installBtn.textContent = '安装中...'; } this.setPluginProgress(plugin.plugin_id, 35, '正在请求服务器安装...'); const response = await fetch(`${top_level_path}/api/plugins/${plugin.plugin_id}/install`, { method: 'POST', credentials: 'same-origin' }); this.setPluginProgress(plugin.plugin_id, 68, '正在处理安装结果...'); const data = await response.json(); if (data.success) { plugin.installed = true; plugin.enabled = true; plugin.installed_version = plugin.version; this.setPluginProgress(plugin.plugin_id, 82, '正在加载插件资源...'); if (window.pluginLoader && (data.plugin?.entry_point || plugin.entry_point)) { if (data.plugin) { plugin.entry_point = data.plugin.entry_point || plugin.entry_point; plugin.styles = data.plugin.styles || plugin.styles; } await window.pluginLoader.load(plugin); } this.setPluginProgress(plugin.plugin_id, 100, '安装完成', 'success'); showToast(`插件 "${plugin.name}" 安装成功`, 'success'); setTimeout(() => { this.renderPluginList(); this.updateBatchButtons(false); }, 500); } else { this.setPluginProgress( plugin.plugin_id, 100, `安装失败：${data.message || '未知错误'}`, 'error' ); if (installBtn) { installBtn.disabled = false; installBtn.textContent = '安装'; } showToast(`安装失败: ${data.message || '未知错误'}`, 'error'); } } catch (error) { this.setPluginProgress( plugin.plugin_id, 100, `安装失败：${error.message}`, 'error' ); if (installBtn) { installBtn.disabled = false; installBtn.textContent = '安装'; } showToast(`安装失败: ${error.message}`, 'error'); } }
